// Vercelのサーバー機能が正しく動作しているかを確認するための「診断用コード」です。

// Googleの認証は一切行いません。


export default async function handler(req, res) {

  // CORSヘッダーを設定

  res.setHeader('Access-Control-Allow-Origin', '*');

  res.setHeader('Access-Control-Allow-Methods', 'POST, GET, OPTIONS');

  res.setHeader('Access-Control-Allow-Headers', 'Content-Type');


  // preflight requestへの対応

  if (req.method === "OPTIONS") {

    return res.status(204).send("");

  }

  

  // GETリクエストが来たら、成功のメッセージを返す

  if (req.method === 'GET') {

    console.log("GETリクエストを受け取りました。テストは成功です。");

    return res.status(200).json({ status: "ok", message: "バックエンドは正常に動作しています！" });

  }


  // POSTリクエストが来ても、今はテスト中なので簡単なメッセージを返す

  if (req.method === 'POST') {

     console.log("POSTリクエストを受け取りました。テストは成功です。");

     return res.status(200).json({ status: "ok", message: "バックエンドはPOSTリクエストを正常に受け取りました！" });

  }


  // それ以外のリクエストは拒否

  return res.status(405).json({ error: "Method Not Allowed" });

}
